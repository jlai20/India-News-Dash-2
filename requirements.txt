<script>
const IST_TZ = "Asia/Kolkata";
let DATA = null;
let activeCat = "all";
let activeTicker = null;
let query = "";
let winOnly = true;

const $ = s => document.querySelector(s);

/* ---- IST helpers ---- */
function istParts(d=new Date()){
  const f = new Intl.DateTimeFormat("en-GB",{timeZone:IST_TZ,hour12:false,
    weekday:"short",day:"2-digit",month:"short",hour:"2-digit",minute:"2-digit",second:"2-digit"});
  const p = {};
  for(const x of f.formatToParts(d)) p[x.type]=x.value;
  return p;
}
function istMinutes(d=new Date()){
  const p=istParts(d); return (+p.hour)*60 + (+p.minute);
}
function istWeekday(d=new Date()){ // 0=Sun..6=Sat
  const wd=new Intl.DateTimeFormat("en-US",{timeZone:IST_TZ,weekday:"short"}).format(d);
  return ["Sun","Mon","Tue","Wed","Thu","Fri","Sat"].indexOf(wd);
}

/* ---- live clock + market status ---- */
function tickClock(){
  const p=istParts();
  $("#clock").innerHTML = `${p.hour}:${p.minute}:${p.second}<small>IST</small>`;
  const mins=istMinutes(), wd=istWeekday();
  const weekday = wd>=1 && wd<=5;
  const open = weekday && mins>=555 && mins<930; // 09:15–15:30
  const sd=$("#mktStatus");
  if(open){ sd.innerHTML='<span class="dot open"></span><span class="open">Market open</span>';
    const left=930-mins; $("#opensIn").textContent=`closes in ${Math.floor(left/60)}h ${left%60}m`;
  } else {
    sd.innerHTML='<span class="dot closed"></span><span class="closed">Market closed</span>';
    $("#opensIn").textContent = `opens in ${untilOpenStr()}`;
  }
  paintTimeline();
}
function nextOpen(){ // Date of next 09:15 IST weekday
  let d=new Date();
  for(let i=0;i<8;i++){
    const wd=istWeekday(d), mins=istMinutes(d);
    if(wd>=1 && wd<=5 && (i>0 || mins<555)) {
      // build that day's 09:15 IST as an instant
      return istInstant(d,9,15);
    }
    d=new Date(d.getTime()+24*3600*1000);
  }
  return istInstant(d,9,15);
}
function istInstant(refDate,h,m){
  // returns a Date (UTC instant) for given IST clock time on refDate's IST calendar day
  const p=istParts(refDate);
  // reconstruct yyyy from a full format
  const y=new Intl.DateTimeFormat("en-US",{timeZone:IST_TZ,year:"numeric"}).format(refDate);
  const moNum={Jan:0,Feb:1,Mar:2,Apr:3,May:4,Jun:5,Jul:6,Aug:7,Sep:8,Oct:9,Nov:10,Dec:11}[p.month];
  // IST = UTC+5:30 → UTC instant
  return new Date(Date.UTC(+y,moNum,+p.day,h-5,m-30,0));
}
function untilOpenStr(){
  const ms=nextOpen()-new Date();
  if(ms<0) return "—";
  const h=Math.floor(ms/3600000), m=Math.floor(ms%3600000/60000);
  return h>=24 ? `${Math.floor(h/24)}d ${h%24}h` : `${h}h ${m}m`;
}
function paintTimeline(){
  if(!DATA) return;
  const close=new Date(DATA.window.since_utc).getTime();
  const open=nextOpen().getTime();
  const now=Date.now();
  const span=open-close;
  let frac = span>0 ? (now-close)/span : 0;
  frac=Math.max(0,Math.min(1,frac));
  $("#tlFill").style.width=(frac*100).toFixed(1)+"%";
  $("#tlNow").style.left=(frac*100).toFixed(1)+"%";
}

/* ---- relative time ---- */
function ago(iso){
  if(!iso) return "";
  const m=Math.floor((Date.now()-new Date(iso))/60000);
  if(m<1) return "just now";
  if(m<60) return m+"m ago";
  const h=Math.floor(m/60);
  if(h<24) return h+"h ago";
  return Math.floor(h/24)+"d ago";
}
function esc(s){return (s||"").replace(/[&<>"]/g,c=>({"&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;"}[c]));}

/* ---- render ---- */
function filtered(){
  if(!DATA) return [];
  const q=query.trim().toLowerCase();
  return DATA.items.filter(it=>{
    if(winOnly && !it.in_window) return false;
    if(activeCat!=="all" && !it.categories.includes(activeCat)) return false;
    if(activeTicker && !it.tickers.some(t=>t.ticker===activeTicker)) return false;
    if(q){
      const hay=(it.title+" "+it.summary+" "+it.sources.join(" ")+" "+
        it.tickers.map(t=>t.ticker+" "+t.name).join(" ")).toLowerCase();
      if(!hay.includes(q)) return false;
    }
    return true;
  });
}
function render(){
  const items=filtered();
  const feed=$("#feed");
  if(!items.length){
    feed.innerHTML=`<div class="empty"><div class="big">Nothing here yet</div>
      <div>No stories match this filter${winOnly?` in the overnight window`:``}.
      ${DATA&&DATA.seed?`This is placeholder data — run the GitHub Action to load live news.`:`Try “All”, clear filters, or untick “overnight only”.`}</div></div>`;
    return;
  }
  feed.innerHTML=items.map(it=>{
    const cats=it.categories.map(c=>`<span class="tag ${c}">${c==="fno"?"F&amp;O":c}</span>`).join("");
    const chips=it.tickers.map(t=>`<span class="chip" data-tk="${t.ticker}" title="${esc(t.name)}">${t.ticker}</span>`).join("");
    const srcs=it.sources.map(s=>`<span>${esc(s)}</span>`).join("");
    return `<article class="item">
      <div class="meta">
        <span class="time">${esc(it.published_ist||"")}</span>
        <span class="ago">${ago(it.published_utc)}</span>
        <span class="srcs">${srcs}</span>
      </div>
      <h2><a href="${esc(it.link)}" target="_blank" rel="noopener">${esc(it.title)}</a></h2>
      ${it.summary?`<p>${esc(it.summary)}</p>`:""}
      <div class="tags">${cats}${chips}</div>
    </article>`;
  }).join("");
  feed.querySelectorAll(".chip").forEach(c=>c.onclick=()=>{
    activeTicker = activeTicker===c.dataset.tk ? null : c.dataset.tk;
    syncStockHighlight(); render();
    window.scrollTo({top:0,behavior:"smooth"});
  });
}
function renderRail(){
  const sl=$("#stockList");
  if(DATA.top_stocks.length){
    sl.innerHTML=DATA.top_stocks.map(s=>`
      <div class="stockrow ${activeTicker===s.ticker?"active":""}" data-tk="${s.ticker}">
        <div class="left"><span class="tk">${s.ticker}</span><span class="nm">${esc(s.name)}</span></div>
        <span class="ct">${s.count}</span>
      </div>`).join("");
    sl.querySelectorAll(".stockrow").forEach(r=>r.onclick=()=>{
      activeTicker = activeTicker===r.dataset.tk ? null : r.dataset.tk;
      syncStockHighlight(); render();
    });
  } else { sl.innerHTML=`<div style="color:var(--faint);font-size:12px">No F&O names in window yet.</div>`; }

  $("#srcGrid").innerHTML=DATA.sources.map(s=>`
    <div class="srcrow"><span class="sd ${s.ok?"ok":"bad"}"></span>
      <span>${esc(s.name)}</span><span class="sc">${s.ok?s.count:"—"}</span></div>`).join("");
}
function syncStockHighlight(){
  document.querySelectorAll(".stockrow").forEach(r=>
    r.classList.toggle("active", r.dataset.tk===activeTicker));
}
function renderCounts(){
  const c=DATA.counts;
  $("#cAll").textContent=winOnly?c.in_window:c.total_fetched;
  $("#cMacro").textContent=c.macro;
  $("#cFno").textContent=c.fno;
}
function renderHeader(){
  $("#winLabel").textContent=DATA.window.label.replace(/^Since previous close — /,"");
  $("#tlClose").textContent=DATA.window.since_ist.split(" ")[0].slice(5); // MM-DD-ish; small
  const no=nextOpen();
  $("#tlOpen").textContent=new Intl.DateTimeFormat("en-GB",{timeZone:IST_TZ,weekday:"short",day:"2-digit",month:"short"}).format(no);
  $("#updated").textContent="Updated "+(DATA.generated_at_ist||"");
  $("#genline").textContent=DATA.generated_at_ist?("Last refresh: "+DATA.generated_at_ist+"."):"";
  $("#seedFlag").innerHTML = DATA.seed ? `<span class="seedflag">sample data</span>` : "";
}

/* ---- data load ---- */
async function load(){
  try{
    const r=await fetch("./data/news.json?t="+Date.now(),{cache:"no-store"});
    if(!r.ok) throw new Error(r.status);
    DATA=await r.json();
    renderHeader(); renderCounts(); renderRail(); render(); paintTimeline();
  }catch(e){
    $("#feed").innerHTML=`<div class="empty"><div class="big">Couldn’t load the feed</div>
      <div>data/news.json isn’t available yet. If you just deployed, run the
      “Update news” GitHub Action once, then refresh.</div></div>`;
  }
}

/* ---- wire controls ---- */
$("#catSeg").addEventListener("click",e=>{
  const b=e.target.closest("button"); if(!b) return;
  activeCat=b.dataset.cat;
  [...$("#catSeg").children].forEach(x=>{
    x.classList.remove("on","all","macro","fno");
    if(x===b){ x.classList.add("on"); x.classList.add(b.dataset.cat); }
  });
  render();
});
